Neues Projekt anlagen
	cd <projektverzeichnis>
	rails new <projekt_name>
	cd <projekt_name>

Server starten:
	rails s

App anschauen
	localhost:3000

Ordnerstruktur:
	app
		#hier liegen die entscheidenen Dateien
		#Unterordner nach dem MVC-Prinzip
		(ModelViewController: Die Datenhaltung ist von der Logik un von der Darstellung möglichst strikt zu trennen)
		models
			#Klassen für die Tabellen
		controllers
			#Klassen für die Logik und Abwicklung eines Requests
		views
			#Unterordner mit den Dateien der Darstellung
